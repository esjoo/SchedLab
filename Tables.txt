CREATE TABLE Users (
    UserNum int NOT NULL PRIMARY KEY AUTO_INCREMENT, //Auto inc?
    Lab varchar(255),
    Pword varchar(255) NOT NULL,
    Name varchar(255) NOT NULL, 
    Manage_Num int //Foreign key?
    
);

CREATE TABLE Make (
    UserNum int,
    ProtocolNum int,
    Time int //time in hours? 
);

CREATE TABLE Include (
    ProtocolNum int,
    SupID int,
    Dosage varchar(255) //Dosage in varchar? 
);

CREATE TABLE Consume (
    ExpeNum int,
    SupID int,
    List text //text? 
);


ALTER TABLE Make
    ADD (FOREIGN KEY (UserNum) REFERENCES Users(UserNum),
        FOREIGN KEY (ProtocolNum) REFERENCES Protocol(ProtocolNum)); 

ALTER TABLE Include
    ADD (FOREIGN KEY (ProtocolNum) REFERENCES Protocol(ProtocolNum),
        FOREIGN KEY (SupID) REFERENCES Supplement(SupID)); 

ALTER TABLE Consume
